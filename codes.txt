This is a **Phase 2 Feature**, which means we first need to add the `Homework` and `HomeworkSubmission` models to your Prisma schema before the UI can save or fetch data.

Here is the complete synchronization for the Homework & Assignments module, including the database update, server actions, and the Next.js UI page.

### 1. Update Database Schema (`schema.prisma`)

Add these models to your `prisma/schema.prisma` file and run `npx prisma db push`.

```prisma
model Homework {
  id          String   @id @default(cuid())
  title       String
  description String?
  dueDate     DateTime
  attachmentUrl String? // For Cloudinary/S3 links
  
  subjectId   String
  subject     Subject  @relation(fields: [subjectId], references: [id])
  classId     String
  class       Class    @relation(fields: [classId], references: [id])
  teacherId   String
  teacher     Staff    @relation(fields: [teacherId], references: [id])

  submissions HomeworkSubmission[]
  createdAt   DateTime @default(now())
}

model HomeworkSubmission {
  id          String   @id @default(cuid())
  homeworkId  String
  homework    Homework @relation(fields: [homeworkId], references: [id])
  studentId   String
  student     Student  @relation(fields: [studentId], references: [id])
  
  fileUrl     String?  // Student's uploaded work
  status      String   @default("PENDING_REVIEW") // PENDING_REVIEW, GRADED
  marks       Float?
  feedback    String?
  
  submittedAt DateTime @default(now())
  
  @@unique([homeworkId, studentId])
}

```

### 2. Create the Server Action

Create a new file at `app/actions/homeworkActions.ts` to handle the form submission.

```typescript
// app/actions/homeworkActions.ts
'use server';

import prisma from '@/lib/prisma';
import { revalidatePath } from 'next/cache';
import { getServerSession } from "next-auth/next";
import { authOptions } from "@/app/api/auth/[...nextauth]/route";

export async function createHomework(formData: FormData) {
  const session = await getServerSession(authOptions);
  const teacherId = (session?.user as any).id;

  const title = formData.get('title') as string;
  const subjectId = formData.get('subjectId') as string;
  const dueDate = formData.get('dueDate') as string;
  const description = formData.get('description') as string;
  // const file = formData.get('attachment') as File; // TODO: Upload to Cloudinary if needed

  // Get the teacher's profile and find the class associated with this subject
  const teacher = await prisma.staff.findUnique({ where: { userId: teacherId } });
  const subject = await prisma.subject.findUnique({ where: { id: subjectId } });

  if (!teacher || !subject) throw new Error("Invalid teacher or subject");

  await prisma.homework.create({
    data: {
      title,
      description,
      dueDate: new Date(dueDate),
      subjectId,
      classId: subject.classId,
      teacherId: teacher.id,
    }
  });

  revalidatePath('/teacher/academics');
}

```

### 3. The Teacher Homework Page (`app/teacher/academics/page.tsx`)

This syncs your provided Teacher Dashboard HTML into a dynamic Next.js page. It fetches the active assignments and recent submissions, then maps them into your Bento Grid layout.

```tsx
// app/teacher/academics/page.tsx
import { getServerSession } from "next-auth/next";
import { authOptions } from "@/app/api/auth/[...nextauth]/route";
import prisma from "@/lib/prisma";
import { createHomework } from "@/app/actions/homeworkActions";

export const dynamic = 'force-dynamic';

export default async function TeacherAcademicsPage() {
  const session = await getServerSession(authOptions);
  
  const teacher = await prisma.staff.findUnique({
    where: { userId: (session?.user as any).id },
    include: {
      subjects: { include: { class: true } }
    }
  });

  if (!teacher) return <div>Teacher profile not found.</div>;

  // Fetch Active Assignments created by this teacher
  const assignments = await prisma.homework.findMany({
    where: { teacherId: teacher.id },
    include: {
      subject: true,
      class: true,
      submissions: true,
      _count: { select: { submissions: true } }
    },
    orderBy: { createdAt: 'desc' }
  });

  // Calculate total students in the classes this teacher teaches for the progress bars
  const classIds = teacher.subjects.map(s => s.classId);
  const totalStudentsInClasses = await prisma.student.count({
    where: { classId: { in: classIds }, status: 'ACTIVE' }
  });

  // Fetch recent submissions across all this teacher's assignments
  const recentSubmissions = await prisma.homeworkSubmission.findMany({
    where: { homework: { teacherId: teacher.id } },
    include: { student: true, homework: true },
    orderBy: { submittedAt: 'desc' },
    take: 10
  });

  return (
    <main className="md:ml-64 min-h-screen flex flex-col bg-surface">
      {/* TopAppBar Shell */}
      <header className="sticky top-0 z-40 bg-surface shadow-sm px-lg py-sm flex justify-between items-center w-full border-b border-border-default">
        <div className="flex items-center gap-md">
          <span className="font-h3 text-h3 font-bold text-school-blue-dark">St. Mary's Academy ERP</span>
          <nav className="hidden md:flex ml-xl gap-lg">
            <span className="text-primary font-bold border-b-2 border-primary font-label text-label py-4">Homework</span>
          </nav>
        </div>
      </header>

      <div className="flex-1 p-page-desktop space-y-section max-w-container-max mx-auto w-full text-on-surface">
        <section className="grid grid-cols-12 gap-lg">
          
          {/* Create Homework Form (Featured Card) */}
          <div className="col-span-12 lg:col-span-5 bg-white rounded-lg border-2 border-school-blue-dark p-8 shadow-md">
            <div className="flex items-center gap-md mb-lg">
              <div className="p-3 bg-school-blue-light text-school-blue-dark rounded-xl">
                <span className="material-symbols-outlined">add_task</span>
              </div>
              <h2 className="font-h2 text-h2 text-school-blue-dark">New Assignment</h2>
            </div>
            
            <form action={createHomework} className="space-y-lg">
              <div className="space-y-xs">
                <label className="font-label text-label text-on-surface-variant">Homework Title</label>
                <input name="title" required className="w-full px-4 py-3 rounded-lg border border-border-default focus:ring-2 focus:ring-school-blue-dark outline-none transition-all" placeholder="e.g. Calculus: Limits & Continuity" type="text"/>
              </div>
              
              <div className="grid grid-cols-2 gap-md">
                <div className="space-y-xs">
                  <label className="font-label text-label text-on-surface-variant">Subject</label>
                  <select name="subjectId" required className="w-full px-4 py-3 rounded-lg border border-border-default focus:ring-2 focus:ring-school-blue-dark outline-none bg-white">
                    {teacher.subjects.map(sub => (
                      <option key={sub.id} value={sub.id}>{sub.name} ({sub.class.name}-{sub.class.section})</option>
                    ))}
                  </select>
                </div>
                <div className="space-y-xs">
                  <label className="font-label text-label text-on-surface-variant">Due Date</label>
                  <input name="dueDate" required className="w-full px-4 py-3 rounded-lg border border-border-default focus:ring-2 focus:ring-school-blue-dark outline-none bg-white" type="datetime-local"/>
                </div>
              </div>

              <div className="space-y-xs">
                <label className="font-label text-label text-on-surface-variant">Assignment Details</label>
                <textarea name="description" className="w-full px-4 py-3 rounded-lg border border-border-default focus:ring-2 focus:ring-school-blue-dark outline-none" placeholder="Describe the homework objectives and requirements..." rows={3}></textarea>
              </div>

              <button type="submit" className="w-full py-4 bg-school-blue-dark text-white rounded-lg font-bold shadow-md transition-all active:scale-95 flex items-center justify-center gap-md">
                <span className="material-symbols-outlined">send</span>
                Publish Assignment
              </button>
            </form>
          </div>

          {/* Grading & Submissions (Bento Main Card) */}
          <div className="col-span-12 lg:col-span-7 space-y-lg">
            
            {/* Submissions List */}
            <div className="bg-white rounded-lg border border-border-default overflow-hidden flex flex-col h-[520px]">
              <div className="p-6 border-b border-border-default flex justify-between items-center bg-bg-secondary">
                <div>
                  <h3 className="font-h3 text-h3 text-on-surface">Recent Submissions</h3>
                  <p className="font-body-small text-body-small text-on-surface-variant">Across all your active assignments</p>
                </div>
              </div>
              
              <div className="flex-1 overflow-y-auto custom-scrollbar">
                <table className="w-full text-left border-collapse">
                  <thead className="sticky top-0 bg-white z-10">
                    <tr className="font-overline text-overline text-on-surface-variant uppercase bg-surface-container-low border-b border-border-default">
                      <th className="px-6 py-3">Student Name</th>
                      <th className="px-6 py-3">Assignment</th>
                      <th className="px-6 py-3">Status</th>
                      <th className="px-6 py-3 text-right">Action</th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-border-default">
                    {recentSubmissions.length === 0 ? (
                      <tr><td colSpan={4} className="text-center p-6 text-text-tertiary">No recent submissions.</td></tr>
                    ) : (
                      recentSubmissions.map(sub => (
                        <tr key={sub.id} className="hover:bg-school-blue-extra-light transition-colors group cursor-pointer">
                          <td className="px-6 py-4">
                            <div className="flex items-center gap-md">
                              <span className="font-label text-label font-bold text-on-surface">{sub.student.name}</span>
                            </div>
                          </td>
                          <td className="px-6 py-4 font-body-small text-body-small text-on-surface-variant">
                            {sub.homework.title}
                          </td>
                          <td className="px-6 py-4">
                            {sub.status === 'GRADED' ? (
                              <span className="px-3 py-1 rounded-full bg-teal-light text-teal-dark font-label text-caption font-bold">Graded</span>
                            ) : (
                              <span className="px-3 py-1 rounded-full bg-amber-light text-amber-dark font-label text-caption font-bold">Pending Review</span>
                            )}
                          </td>
                          <td className="px-6 py-4 text-right">
                            {sub.status === 'PENDING_REVIEW' ? (
                              <button className="bg-school-blue-dark text-white px-3 py-1 rounded-lg text-caption font-bold hover:bg-school-blue-light hover:text-school-blue-dark transition-all">Grade Now</button>
                            ) : (
                              <span className="font-bold text-teal-dark">{sub.marks} Pts</span>
                            )}
                          </td>
                        </tr>
                      ))
                    )}
                  </tbody>
                </table>
              </div>
            </div>
          </div>
        </section>

        {/* Bottom Section: Active Assignments Gallery */}
        <section className="space-y-lg mt-8">
          <div className="flex justify-between items-end">
            <div>
              <h2 className="font-h2 text-h2 text-on-surface">Active Assignments</h2>
              <p className="font-body text-body text-on-surface-variant">Current homework being tracked across all your classes.</p>
            </div>
          </div>
          
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-lg">
            {assignments.map(hw => {
              // Calculate completion percentage
              // For a real app, you'd calculate students per class rather than totalStudentsInClasses.
              // We use a simplified fallback here for the MVP visual.
              const maxStudents = 30; // Mock class size
              const pct = Math.round((hw._count.submissions / maxStudents) * 100);

              return (
                <div key={hw.id} className="bg-white rounded-lg border border-border-default p-6 hover:shadow-lg transition-all group">
                  <div className="flex justify-between mb-md">
                    <span className="px-2 py-1 rounded bg-teal-light text-teal-dark font-bold text-[10px] uppercase tracking-wider">
                      {hw.subject.name}
                    </span>
                    <span className="material-symbols-outlined text-outline">more_vert</span>
                  </div>
                  <h4 className="font-h4 text-h4 text-on-surface mb-sm">{hw.title}</h4>
                  <div className="flex items-center gap-sm text-on-surface-variant font-caption text-caption mb-lg">
                    <span className="material-symbols-outlined text-sm">event</span>
                    Due: {new Date(hw.dueDate).toLocaleDateString()}
                  </div>
                  <div className="w-full bg-bg-tertiary h-2 rounded-full overflow-hidden mb-xs">
                    <div className="bg-teal-dark h-full" style={{ width: `${pct > 100 ? 100 : pct}%` }}></div>
                  </div>
                  <div className="flex justify-between items-center font-caption text-caption text-on-surface-variant">
                    <span>{pct}% Submissions</span>
                    <span className="font-bold text-teal-dark">{hw._count.submissions} Submitted</span>
                  </div>
                </div>
              );
            })}
          </div>
        </section>
      </div>
    </main>
  );
}
```
